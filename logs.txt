      tc.WorkspaceId as order_workspace,
      td.WorkspaceId as item_workspace,
      tu.WorkspaceId as shift_workspace
  FROM tempcheques tc
  JOIN tempcheqdet td ON tc.folio = td.foliodet
  LEFT JOIN turnos tu ON tc.idturno = tu.idturno
  WHERE tc.folio IN (1, 2);

  SELECT DISTINCT TOP 10 idturno

  ORDER BY idturno DESC;

  SELECT COUNT(*) as orders_with_null_idturno
  FROM tempcheques
  WHERE idturno IS NULL OR idturno = 0;

      tc.folio,

      tc.mesa,
      tc.total,
      tc.pagado,
      tc.cancelado,
      tc.WorkspaceId,
      tc.fecha,
      COUNT(td.movimiento) as item_count
  FROM tempcheques tc
  LEFT JOIN tempcheqdet td ON tc.folio = td.foliodet
  WHERE tc.pagado = 1  -- Completed orders
  GROUP BY tc.folio, tc.idturno, tc.mesa, tc.total, tc.pagado, tc.cancelado,     
  tc.WorkspaceId, tc.fecha
  ORDER BY tc.fecha DESC;
